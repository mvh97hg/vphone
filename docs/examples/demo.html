💻 Code Examples

Simple Floating Widget

<!-- Load library -->
<script src="lib/vphone.js"></script>

<!-- Mount widget -->
<script>
  var phone = Vphone.mount({
    phoneUrl: 'phone/index.html',
    floating: true,
    startCollapsed: true,
    expandOnIncomingCall: true,
    config: {
      Language: 'en',
      WebSocketPort: 5066,
      wssServer: 'sip.example.com'
    }
  });
</script>

Inline Embedding with Container

<!-- Container -->
<div id="phone-widget"></div>

<!-- Load library -->
<script src="lib/vphone.js"></script>

<!-- Mount phone -->
<script>
  var phone = Vphone.mount({
    container: '#phone-widget',
    phoneUrl: 'phone/index.html',
    width: '400px',
    height: '600px',
    config: {
      Language: 'en'
    }
  });
</script>

Advanced Configuration & Events

var phone = Vphone.mount({
  phoneUrl: 'phone/index.html',
  floating: true,
  
  // Event callbacks
  onStatusChange: function(status) {
    console.log('Registered:', status.registered);
    updateUI(status.text);
  },
  
  onCallEvent: function(event) {
    switch(event.phase) {
      case 'incoming':
        phone.expand();
        break;
      case 'ended':
        logCall(event);
        break;
    }
  },
  
  // Phone config
  config: {
    Language: 'en',
    WebSocketPort: 5066,
    wssServer: 'sip.example.com',
  }
});

// Programmatic control
phone.expand();
phone.setStatus(true, 'Available');
